自定义对话框
html：
<div id="mydialog" style="display:none;">
	<div>
	商品名称：<input type="text"><br>
	商品品类：<input type="text"><br>
	<button class="btn-close">关闭对话框</button>
	</div>
</div>	
js：
$('#mydialog').on('click','.btn-close',function(){
	$('#mydialog').dialog().hide();
});
$('#mydialog').dialog('对话框标题',function(){
	//点击确认后的操作
	var inputs=$(this).find('input');
	$.alert(inputs.eq(0).val() + '  ' + inputs.eq(1).val());
},function(){
	//点击取消后的操作
	console.log('cancel');
	//如果只需要一个按钮，下面buttons方法只传入一个参数
}).buttons('自定义确认','自定义取消');
自定义列表选择对话框
<script type="text/javascript" src="/ug/b/js/list_dlg.js"></script>
var conf = {
	title : '请选择XXX',
	url : '/Admin/Index/test',
	setData : function(jo){
		var data = jo.rows;
		var html = '';
		html += '<ul>';
		for (var i = 0; i < data.length; i++) {
			html += '<li class="myitem"' + this.getNoHtml(i) + '>' + data[i].c0 + '--' + data[i].c1 + '--' + data[i].c2 + '--' + '</li>';
		}
		for (var i = 0; i < data.length; i++) {
			html += '<li class="myitem"' + this.getNoHtml(i) + '>' + data[i].c0 + '--' + data[i].c1 + '--' + data[i].c2 + '--' + '</li>';
		}
		html += '</ul>';				
		return {data:data,html:html}
	},
	event : {
		cls : 'myitem',
		callback : function(d){
			console.log(d);
			$.alert(d.c0);
		}
	},
	button : {
		title : '哈哈哈',
		callback : function(){
			console.log('click button');
		}
	},
};
$.listDialog(conf);

$.listDialog().show();
